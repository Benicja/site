  - item: "Red Pepper"
    amount: "1"
  - item: "Sweet Potato"
    amount: "1"
  - item: "Head Broccoli"
    amount: "1"
  - item: "Olive Oil"
    amount: "as needed"
  - item: "Gnocchi or Rice"
    amount: "500g gnocchi or 200g rice"
  - item: "Can Chopped Tomatoes"
    amount: "400g"
  - item: "Mixed Beans"

  - item: "Sweetcorn"

  - item: "Smoked Paprika"
    amount: "1 tsp"
  - item: "Salt"
    amount: "to taste"
  - item: "Pepper"
    amount: "to taste"
instructions:
  - step: "Prepare a saucepan."
  - step: "Steam cook Broccoli until just tender; set aside."
  - step: "Dice Onion, Garlic, Sweet Potato, Red Pepper. Fry in Olive Oil in large pan."
  - step: "Fry 5 minutes, add Smoked Paprika, Salt, Pepper. Stir well."
  - step: "Fry 5 minutes, add Chopped Tomatoes, Mixed Beans, and Sweetcorn."
  - step: "Simmer until all vegetables are soft, about 10 minutes."
  - step: "Add Gnocchi at end, simmer 3–4 minutes. Serve with Broccoli or Rice."
