---
title: "Quiche"
description: "A rich and flavourful quiche Lorraine with vaults of vegetables and cheesy, creamy filling."
featured_image: "/images/recipes/quiche_best.jpg"
prep_time: 30
cook_time: 55
servings: 6
category: "Dinner"
publishDate: 2026-02-10
ingredients:
  - item: "Olive Oil"
    amount: "1 tbsp"
  - item: "Smoked Bacon, Chopped"
    amount: "6 rashers"
  - item: "Eggs"
    amount: "5"
  - item: "Single Cream"
    amount: "284ml"
  - item: "Milk"
    amount: "150ml"
  - item: "Gruyère Cheese, Finely Grated"
    amount: "140g"
  - item: "Bell Pepper"
    amount: "1"
  - item: "Zucchini"
    amount: "1"
  - item: "Small Onion"
    amount: "1"
  - item: "Leek"
    amount: "1"
  - item: "Spinach"
    amount: "50g"
  - item: "Cold Butter (Pastry)"
    amount: "175g"
  - item: "Plain Flour (Pastry)"
    amount: "300g"
  - item: "Egg Yolks (Pastry)"
    amount: "2"
  - item: "Water for Pastry"
    amount: "as needed"
  - item: "Nutmeg"
    amount: "1 tsp"
  - item: "Salt"
    amount: "to taste"
  - item: "Pepper"
    amount: "to taste"
instructions:
  - step: "Mix Flour, Butter, Egg Yolks, add Water to form dough. Chill 10 minutes."
  - step: "Preheat oven to 190°C. Roll pastry, line cake tin, build high wall. Chill 5 minutes."
  - step: "Bake blind 15 minutes."
  - step: "Fry Bacon, Onion, Leek, Garlic, Bell Pepper, Zucchini, Spinach."
  - step: "Beat Eggs, Cream, Milk, Cheese, Nutmeg, Salt, Pepper."
  - step: "Mix veggies and Bacon with Eggs-Cream-Cheese mixture, pour into pastry."
  - step: "Bake for 35–45 minutes. Cool slightly before slicing."
