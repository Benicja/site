ingredients:
  - item: "Digestive Biscuits"
    amount: "225g"
  - item: "Butter, Melted"
    amount: "100g"
  - item: "Tin Caramel or Dulce De Leche"
    amount: "400g"
  - item: "Small Bananas, Sliced"
    amount: "3"
  - item: "Double Cream"
    amount: "300ml"
  - item: "Icing Sugar"
    amount: "1 tbsp"
  - item: "Dark Chocolate, Grated (optional)"
    amount: "10g"
instructions:
  - step: "Crush Biscuits, mix with Butter, press into tin. Chill."
  - step: "Spread Caramel over base."
  - step: "Arrange Bananas."
  - step: "Whip Cream with Icing Sugar, spread over Bananas."
  - step: "Grate Chocolate on top. Chill."
