ingredients:
  - item: "Unsalted Butter, Cubed"
    amount: "200g"
  - item: "Caster Sugar (plus extra for finishing)"
    amount: "100g"
  - item: "Vanilla Extract (optional)"
    amount: "1 tsp"
  - item: "Plain Flour (plus extra for dusting)"
    amount: "300g"
instructions:
  - step: "Preheat oven to 170°C."
  - step: "Mix Butter, Sugar, (Vanilla). Stir in Flour, form dough."
  - step: "Press into tin, prick, sprinkle Sugar."
  - step: "Bake 25–35 min."
