featured_image: "/images/recipes/klakada.jpg"
prep_time: 10
cook_time: 20
servings: 8
category: "Dessert"
publishDate: 2026-02-14
ingredients:
  - item: "Butter for greasing"
    amount: "as needed"
  - item: "Breadcrumbs for dusting"
    amount: "as needed"
  - item: "Butter"
    amount: "100g"
  - item: "Eggs"
    amount: "2"
  - item: "Sugar"
    amount: "250g"
  - item: "Cocoa Powder"
    amount: "3 tbsp"
  - item: "Vanilla Sugar"
    amount: "2 tsp"
  - item: "Plain Flour"
    amount: "150g"
  - item: "Salt"
    amount: "pinch"
  - item: "Powdered Sugar (dusting)"
    amount: "10g"
  - item: "Double Cream (to serve)"
    amount: "200ml"
  - item: "Fresh Berries"
    amount: "as needed"
instructions:
  - step: "Grease and dust tin. Melt Butter, let cool."
  - step: "Whisk Eggs, Sugar, then stir in Butter, Cocoa, Vanilla Sugar, Salt, Flour."
  - step: "Bake at 180°C for 15–20 min."
  - step: "Cool, dust with Sugar, serve with Cream, Berries."
