category: "Dinner"
publishDate: 2026-01-29
ingredients:
  - item: "Onion"
    amount: "1"
  - item: "Garlic Clove"
    amount: "1"
  - item: "Chicken Breast, Diced"
    amount: "500g"
  - item: "Can Chopped Tomatoes"
    amount: "400g"
  - item: "Curry Powder"
    amount: "1 tbsp"
  - item: "Cumin"
    amount: "1 tsp"
  - item: "Salt"
    amount: "1 tsp"
  - item: "Cayenne Pepper"
    amount: "0.5 tsp"
  - item: "Turmeric"
    amount: "0.5 tsp"
  - item: "Paprika"
    amount: "1 tsp"
  - item: "Cinnamon"
    amount: "0.5 tsp"
  - item: "Ginger"
    amount: "1 tsp"
instructions:
  - step: "Fry Onion, then Garlic, and half the spices."
  - step: "Fry Chicken, add remaining spices, cook."
  - step: "Add Tomatoes, add onion mix back, simmer 15 min."
  - step: "Add Salt, adjust seasoning. Serve with Rice."
