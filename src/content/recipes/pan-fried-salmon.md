  - item: "Onion"
    amount: "1"
  - item: "Garlic Cloves"
    amount: "2"
  - item: "Lemon (Zest and Juice)"
    amount: "1"
  - item: "Sweet Potatoes"
    amount: "400g"
  - item: "Salmon Fillets"
    amount: "2"
  - item: "Salt"
    amount: "to taste"
  - item: "Pepper"
    amount: "to taste"
  - item: "Chip Spice (Schwartz) or Mixed Spice"
    amount: "1 tsp"
instructions:
  - step: "Preheat oven to 200°C. Cut Sweet Potatoes into fries, toss with Oil and Chip Spice, bake 25 min."
  - step: "Season Salmon, cook in Oil, skin side down, 5 minutes, then flip for 2–3 minutes."
  - step: "Remove Salmon, sauté Onion and Garlic, add Lemon juice/zest and a little water for sauce."
  - step: "Serve with fries."
